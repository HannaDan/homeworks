function pow (num, degree) {
   var temp;
   var result;
   if (isNaN(num) || isNaN(degree)) { // перевірка на число
      return console.log ('Please use only numbers in the function')
   }
   if (degree === 0) {
     return 1 // будь яке число в ступені 0 буде дорівнювати 1
   } else if (degree > 0) {
     temp = num // присвоювання початкового значення для розрахунку ступеня
     for (let i = 1; i < degree; i++) { // запуск циклу до поки і стане degree - 1
         // console.log(i);
       result = temp *=num; // множення значення temp на num при кожній ітерації
         // console.log (temp)
      }
     return result;
    }
};
